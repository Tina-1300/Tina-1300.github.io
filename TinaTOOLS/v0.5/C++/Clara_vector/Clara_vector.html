DisplayAllTab()

Paramètre : const std::vector<std::string>& vec - Attend un tableau de std::string.

Note :

Cette Methode permet d'afficher tout les valeur du tableau ainsi que leur emplacemnt si il sont a tableau[0] ou tableau[1] etc...

Exemple d'utilisation :

#include <iostream>
#include <vector>
#include "Tina_TOOLS/Clara_vector.hpp"

int main(){
    Clara_vector clara;

    std::vector<std::string> tableau;
    tableau = {"Tina", "Elias", "Noah", "Karis", "Python", "C++", "C", "JAVA", "Farid"};

    clara.DisplayAllTab(tableau);

    return 0; 
}

SeartchTAB()

Paramètre : const std::vector<std::string>& vec - Attend un tableau de std::string.

Paramètre : const std::string search_cara - Attend la chaine de caractère a rechercher dans le tableau de std:string

Note :

Cette Methode permet de chercher dans un tableau de string, une valeur et de retourner sont emplacement si il et a 0, ou 1 etc...

Exemple d'utilisation :

#include <iostream>
#include <vector>
#include "Tina_TOOLS/Clara_vector.hpp"

int main(){
    Clara_vector clara;

    std::vector<std::string> tableau;
    tableau = {"Tina", "Elias", "Noah", "Karis", "Python", "C++", "C", "JAVA", "Farid"};

    std::cout << clara.SeartchTAB(tableau, "Noah"); << std::endl // affichera 2 

    return 0; 
}



DeletTAB()

Paramètre : std::vector<std::string>& vec - Attend un tableau de std::string.

Paramètre : const std::string delet_cara - Attend la chaine de caractère a suprimmer dans le tableau de std:string

Note :

Cette Methode permet de chercher dans un tableau de string, une valeur et de suprimmer la valeur du tableau si la valeur existe bien etc...

Exemple d'utilisation :

#include <iostream>
#include <vector>
#include "Tina_TOOLS/Clara_vector.hpp"

int main(){
    Clara_vector clara;

    std::vector<std::string> tableau;
    tableau = {"Tina", "Elias", "Noah", "Karis", "Python", "C++", "C", "JAVA", "Farid"};
    clara.DisplayTAB(tableau);
    clara.DeletTAB(tableau, "Noah"); // cherche la chaine Noah dans le tableau si elle y et 
    // elle ce fera suprimmer sinon elle renvera -808
    clara.DisplayTAB(tableau);

    return 0; 
}
