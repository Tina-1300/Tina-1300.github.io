98 static std::string Hash_STRING_SHA__(std::string message, size_t choice_hash){
99 std::string text_to_hash = message;
100 unsigned char hash[choice_hash];
101
102 if(choice_hash == SHA256_DIGEST_LENGTH){
103 // Calcul du hachage SHA-256
104 SHA256(reinterpret_cast<const unsigned char*>(text_to_hash.c_str()), text_to_hash.length(), hash);
105 }else if (choice_hash == SHA512_DIGEST_LENGTH){
106 // Calcule du hash SHA-512
107 SHA512(reinterpret_cast<const unsigned char*>(text_to_hash.c_str()), text_to_hash.length(), hash);
108 }else if (choice_hash == SHA_DIGEST_LENGTH){
109 // Calcule du SHA-1
110 SHA1(reinterpret_cast<const unsigned char*>(text_to_hash.c_str()), text_to_hash.length(), hash);
111 }else if (choice_hash == SHA224_DIGEST_LENGTH){
112 // Calcule du SHA-224
113 SHA224(reinterpret_cast<const unsigned char*>(text_to_hash.c_str()), text_to_hash.length(), hash);
114 }else if (choice_hash == SHA384_DIGEST_LENGTH){
115 // Calcule du SHA-384
116 SHA384(reinterpret_cast<const unsigned char*>(text_to_hash.c_str()), text_to_hash.length(), hash);
117 }else{
118 MessageBoxW(NULL, L"Vous avez mal utiliser la fonction", L"Error", MB_ICONERROR | MB_OK);
119 }
120 std::string hash_str = hashToString(hash, choice_hash);
121 return hash_str;
122 }
